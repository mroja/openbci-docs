    def __getitem__(self, p_key):
        if p_key == 'start_timestamp':
            return self.get_start_timestamp()
        elif p_key == 'end_timestamp':
            return self.get_end_timestamp()
        else:
            return self.get_start_tag()[p_key]

        self._end_tag = None
